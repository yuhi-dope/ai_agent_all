## 0. 設計思想

### なぜモジュール間連携が最重要か

```
中小製造業の経営判断は「部分最適の罠」にはまりやすい:
  ・見積は安く出して受注→粗利が低い→設備更新できない→品質低下→失注
  ・設備投資したが稼働率が上がらない→CF悪化→次の投資ができない
  ・材料相場が上がったが値上げ交渉しない→粗利率悪化→赤字転落

  → 16モジュールが「点」で動いても価値は半減
  → モジュール間でデータが流れ、連鎖的にアクションが起こる「面」の価値
  → 1+1=3 のシナジーが、月額¥55万のBPOを¥165万の価値に引き上げる
```

### 2a. ⑨PL → ⑫市場: 材料費比率と材料相場の突合（コードAN）

#### 概要

⑨PLインテリジェンスが算出する「加工種別ごとの材料費比率」と、⑫市場インテリジェンスが収集する「材料相場の変動」を突合し、PLへの具体的なインパクト金額を算出する。

#### 連携ロジック

```python
def calculate_material_impact(
    process_pl: dict,        # ⑨の出力: 加工種別PL
    commodity_changes: dict, # ⑫の出力: 材料相場変動
    material_mapping: dict,  # 加工種別→主要材料のマッピング
) -> list[dict]:
    """材料相場変動のPLインパクトを算出する。"""
    alerts = []

    for process_type, pl in process_pl.items():
        # 加工種別の主要材料を特定
        primary_materials = material_mapping.get(process_type, [])

        for material in primary_materials:
            if material not in commodity_changes:
                continue

            change_rate = commodity_changes[material]["change"]  # 例: 0.08 (+8%)
            material_ratio = pl["material_cost"] / pl["revenue"]  # 例: 0.45

            # PLインパクト = 材料費比率 × 相場変動率
            pl_impact_rate = material_ratio * change_rate  # 例: 0.45 × 0.08 = 0.036
            pl_impact_yen = pl["revenue"] * pl_impact_rate  # 例: 620万 × 3.6% = 22.3万

            # 粗利率への影響
            current_margin = pl["gross_margin"]  # 例: 12.1%
            new_margin = current_margin - (pl_impact_rate * 100)  # 例: 12.1% - 3.6% = 8.5%

            if abs(change_rate) >= ALERT_THRESHOLD:  # 5%超でアラート
                alerts.append({
                    "process_type": process_type,
                    "material": material,
                    "change_rate": change_rate,
                    "material_ratio": material_ratio,
                    "pl_impact_rate": pl_impact_rate,
                    "pl_impact_yen_monthly": int(pl_impact_yen),
                    "margin_before": current_margin,
                    "margin_after": new_margin,
                    "severity": "CRITICAL" if new_margin < 10 else "WARNING",
                    "action": _generate_action(process_type, material, change_rate, new_margin),
                })

    return alerts

ALERT_THRESHOLD = 0.05  # 相場変動5%超でアラート

# 加工種別→主要材料マッピング
PROCESS_MATERIAL_MAPPING = {
    "射出成型":   ["ポリカーボネート", "ABS", "ナイロン", "POM"],
    "CNC旋盤":   ["SS400", "S45C", "SUS304", "A5052", "C3604"],
    "MCセンタ":  ["SS400", "S45C", "SUS304", "A5052", "A7075"],
    "板金":      ["SPCC", "SUS304", "A5052"],
    "プレス":    ["SPCC", "SPHC", "SUS304"],
    "表面処理":  ["ニッケル", "クロム", "亜鉛"],
    "溶接":      ["SS400", "SUS304", "溶接ワイヤ"],
    "研磨":      [],  # 材料比率が低いため対象外
    "ワイヤーカット": ["真鍮ワイヤ"],
}
```

#### トリガー条件

| トリガー | 条件 | アクション |
|---|---|---|
| 相場変動アラート | 主要材料の相場変動が±5%超 | PLインパクト算出→Slack通知 |
| 月次PL完了時 | ⑨月次PL完了イベント | 当月の材料費実績と相場推移を突合 |

def classify_tech_investment(
    process_pl_results: list[dict],     # ⑨の出力: 加工種別PL
    positioning_scores: list[dict],     # ⑬の出力: 技術ポジショニングスコア
) -> list[dict]:
    """粗利率×技術成長性の4象限分類を行う。"""

    # 閾値
    MARGIN_THRESHOLD = 0.20       # 粗利率20%を高低の境界
    GROWTH_THRESHOLD = 50         # 成長性スコア50点を高低の境界

    classifications = []

    for pl in process_pl_results:
        process_type = pl["process_type"]
        gross_margin = pl["gross_margin"] / 100  # % → 小数

        # ⑬からポジショニングスコアを取得
        positioning = next(
            (p for p in positioning_scores if p["process_type"] == process_type),
            None
        )
        if not positioning:
            continue

        growth_score = positioning["growth_score"]  # 0-100

        # 4象限分類
        if gross_margin >= MARGIN_THRESHOLD and growth_score >= GROWTH_THRESHOLD:
            quadrant = "STAR"
            action = "投資加速"
            detail = (
                f"{process_type}は高粗利({gross_margin:.1%})+成長技術(スコア{growth_score})。"
                f"⑩設備投資シミュレーターで追加設備の回収計算を推奨"
            )
            priority = 1

        elif gross_margin >= MARGIN_THRESHOLD and growth_score < GROWTH_THRESHOLD:
            quadrant = "CASH_COW"
            action = "現状維持+利益確保"
            detail = (
                f"{process_type}は高粗利({gross_margin:.1%})だが技術成長性は低い(スコア{growth_score})。"
                f"現状維持で利益を確保し、STAR象限への再投資原資に"
            )
            priority = 3

        elif gross_margin < MARGIN_THRESHOLD and growth_score >= GROWTH_THRESHOLD:
            quadrant = "QUESTION_MARK"
            action = "選択と集中"
            detail = (
                f"{process_type}は低粗利({gross_margin:.1%})だが成長技術(スコア{growth_score})。"
                f"チャージレート見直し/自動化投資で改善可能か⑩で試算。改善見込みなしなら撤退"
            )
            priority = 2

        else:
            quadrant = "DOG"
            action = "撤退/縮小検討"
            detail = (
                f"{process_type}は低粗利({gross_margin:.1%})+衰退技術(スコア{growth_score})。"
                f"設備売却/外注化を検討。浮いたリソースをSTAR象限に集中"
            )
            priority = 4

        classifications.append({
            "process_type": process_type,
            "gross_margin": gross_margin,
            "growth_score": growth_score,
            "quadrant": quadrant,
            "action": action,
            "detail": detail,
            "priority": priority,
        })

    return sorted(classifications, key=lambda x: x["priority"])
```

def adjust_location_scores_by_market_trends(
    candidates: list[dict],           # ⑪の候補地リスト
    industry_trends: dict,            # ⑫の業界トレンド（5年予測）
    company_process_types: list[str], # 自社の加工種別
) -> list[dict]:
    """市場トレンドに基づいて立地スコアを動的に調整する。"""

    for candidate in candidates:
        industry_composition = candidate.get("industry_composition", {})
        boost_score = 0.0

        for industry, share in industry_composition.items():
            if industry in industry_trends:
                growth = industry_trends[industry]["growth"]
                avg_growth = (growth[0] + growth[1]) / 2  # 成長率の中央値

                # 成長産業の集積度が高い地域はブースト
                # 衰退産業の集積度が高い地域はペナルティ
                boost_score += share * avg_growth * MARKET_BOOST_WEIGHT

        # サプライチェーンスコア（20点満点）に加減算
        # 最大±5点の調整（スコアの25%を上限）
        adjustment = max(-5, min(5, boost_score * 100))
        candidate["supply_chain_score"] += adjustment
        candidate["total_score"] += adjustment
        candidate["market_trend_adjustment"] = {
            "adjustment_points": round(adjustment, 1),
            "reason": _generate_adjustment_reason(industry_composition, industry_trends),
        }

    # スコア順で再ソート
    return sorted(candidates, key=lambda x: x["total_score"], reverse=True)

def forecast_cashflow(
    pl_data: dict,             # ⑨月次PL
    capex_plans: list[dict],   # ⑩投資計画
    purchase_orders: list[dict],# ⑦発注残
    sales_backlog: list[dict], # ①受注残
    inventory_delta: float,    # ④在庫変動額
    maintenance_forecast: dict,# ⑥修繕費予測
    forecast_months: int = 6,  # 予測期間
) -> dict:
    """6ヶ月先までのCF予測を生成する。"""

    monthly_forecasts = []

    for month in range(1, forecast_months + 1):
        # 営業CF（⑨PLベース）
        operating_cf = (
            pl_data["operating_profit"]
            + pl_data["depreciation"]         # 非現金費用の加算
            - _estimate_tax(pl_data)          # 税金支払
            - inventory_delta / forecast_months  # 運転資本変動
        )

        # 売掛金回収（①受注残ベース）
        ar_collection = sum(
            order["amount"] for order in sales_backlog
            if _collection_month(order) == month
        )

        # 買掛金支払（⑦発注残ベース）
        ap_payment = sum(
            po["amount"] for po in purchase_orders
            if _payment_month(po) == month
        )

        # 投資CF（⑩計画ベース）
        investing_cf = -sum(
            plan["amount"] for plan in capex_plans
            if plan["planned_month"] == month
        )

        # 突発修繕費（⑥予測ベース）
        maintenance_cf = -maintenance_forecast.get(f"month_{month}", 0)

        net_cf = operating_cf + ar_collection - ap_payment + investing_cf + maintenance_cf

        monthly_forecasts.append({
            "month": month,
            "operating_cf": operating_cf,
            "ar_collection": ar_collection,
            "ap_payment": ap_payment,
            "investing_cf": investing_cf,
            "maintenance_cf": maintenance_cf,
            "net_cf": net_cf,
        })

    return {
        "forecasts": monthly_forecasts,
        "cumulative_cf": _cumulative(monthly_forecasts),
        "shortfall_month": _detect_shortfall(monthly_forecasts),  # 資金ショート月の検知
    }

# 売掛金回収条件（製造業の標準）
COLLECTION_TERMS = {

    "大手":    90,       # 末締め翌々月末（90日）
    "現金":    0,        # 即日
}

def build_skill_gap_analysis(
    sop_requirements: list[dict],    # ⑤SOPの技能要件
    equipment_certs: list[dict],     # ⑥設備操作資格
    staff_skills: list[dict],        # 現有人材のスキルDB
    production_plan: dict,           # ②生産計画
) -> dict:
    """技能ギャップ分析を行う。"""

    # 必要スキル一覧（⑤SOP + ⑥設備）
    required_skills = _merge_requirements(sop_requirements, equipment_certs)

    # 現有スキルとのギャップ
    gaps = []
    for skill in required_skills:
        holders = [s for s in staff_skills if skill["skill_id"] in s["skills"]]
        required_count = _calc_required_count(skill, production_plan)

        if len(holders) < required_count:
            gaps.append({
                "skill": skill["name"],
                "required": required_count,
                "current": len(holders),
                "gap": required_count - len(holders),
                "holders": [h["name"] for h in holders],
                "risk": "CRITICAL" if len(holders) <= 1 else "WARNING",
                "mitigation": _suggest_mitigation(skill, holders),
            })

    # 退職リスク（年齢×勤続年数ベース）
    retirement_risks = _assess_retirement_risk(staff_skills)

    return {
        "skill_gaps": gaps,
        "retirement_risks": retirement_risks,
        "succession_pairs": _suggest_succession_pairs(gaps, staff_skills),
    }

def _assess_retirement_risk(staff_skills: list[dict]) -> list[dict]:
    """退職リスクを評価する（年齢/勤続年数ベース）。"""
    risks = []
    for staff in staff_skills:
        age = staff.get("age", 0)
        tenure = staff.get("tenure_years", 0)
        unique_skills = staff.get("unique_skills", [])  # この人しかできない技能

        risk_score = 0
        if age >= 60: risk_score += 40
        elif age >= 55: risk_score += 25
        elif age >= 50: risk_score += 10

        if len(unique_skills) > 0: risk_score += 30  # 属人的スキル
        if tenure >= 30: risk_score += 15  # 長期勤続（退職時のインパクト大）

        if risk_score >= 50:
            risks.append({
                "staff_name": staff["name"],
                "age": age,
                "risk_score": risk_score,
                "severity": "CRITICAL" if risk_score >= 70 else "WARNING",
                "unique_skills": unique_skills,
                "succession_status": staff.get("successor", "未定"),
            })

    return sorted(risks, key=lambda x: x["risk_score"], reverse=True)

def calculate_enterprise_value(
    pl_data: dict,               # ⑨EBITDA
    equipment_valuations: list,  # ⑩⑥設備の時価評価
    tech_portfolio: dict,        # ⑬技術資産評価
    cashflow_forecast: dict,     # ⑭将来CF予測
    hr_assessment: dict,         # ⑮人材評価
) -> dict:
    """3手法で企業価値を算出し、総合判断を提示する。"""

    # 手法1: 純資産法（時価ベース）
    tangible_assets = sum(e["market_value"] for e in equipment_valuations)
    intangible_premium = _calc_tech_premium(tech_portfolio)  # 技術力プレミアム
    net_asset_value = tangible_assets + intangible_premium - total_liabilities

    # 手法2: DCF法（⑭のCF予測を使用）
    annual_fcf = cashflow_forecast["annual_free_cashflow"]
    growth_rate = cashflow_forecast["fcf_growth_rate"]
    discount_rate = 0.08  # 中小製造業のWACC（リスクプレミアム込み）
    terminal_value = annual_fcf * (1 + growth_rate) / (discount_rate - growth_rate)
    dcf_value = sum(
        fcf / (1 + discount_rate) ** t
        for t, fcf in enumerate(cashflow_forecast["annual_fcfs"], 1)
    ) + terminal_value / (1 + discount_rate) ** len(cashflow_forecast["annual_fcfs"])

    # 手法3: EBITDA倍率法（中小製造業の相場: 3-6倍）
    ebitda = pl_data["ebitda"]
    ebitda_multiple = _determine_multiple(tech_portfolio, hr_assessment)  # 3-6倍
    ebitda_value = ebitda * ebitda_multiple

    # 承継リスク評価（⑮人材データ）
    succession_risk = _evaluate_succession_risk(hr_assessment)

    return {
        "net_asset_value": net_asset_value,
        "dcf_value": dcf_value,
        "ebitda_value": ebitda_value,
        "recommended_range": (
            min(net_asset_value, dcf_value, ebitda_value),
            max(net_asset_value, dcf_value, ebitda_value),
        ),
        "succession_risk": succession_risk,
        "action_plan": _generate_succession_action_plan(succession_risk, hr_assessment),
    }

# EBITDA倍率の決定要因
def _determine_multiple(tech_portfolio, hr_assessment):
    """技術力と人材の安定性でEBITDA倍率を調整する。"""
    base = 4.0  # 中小製造業の中央値

    # 技術力プレミアム
    if tech_portfolio.get("avg_cpk", 0) >= 1.50:
        base += 0.5  # 高精度加工
    if "5軸MC" in tech_portfolio.get("equipment_types", []):
        base += 0.5  # 高付加価値設備

    # 人材リスクディスカウント
    if hr_assessment.get("skill_dependency_score", 0) >= 0.80:
        base -= 1.0  # 属人性が高い
    if hr_assessment.get("avg_successor_readiness", 0) < 0.30:
        base -= 0.5  # 後継者が育っていない

    return max(3.0, min(6.0, base))
```

if pipeline_name == "manufacturing/pl_intelligence":

        anomaly_alerts = result.final_output.get("anomaly_alerts", [])
        margin_alerts = [

        ]
        if margin_alerts:
            asyncio.create_task(

            )
            triggered.append("manufacturing/margin_decline_chain")

async def _run_margin_decline_chain(company_id: str, margin_alerts: list) -> None:
    """粗利率急落チェーン: ⑫材料相場チェック → 値上げ提案生成。"""
    try:
        from workers.bpo.manufacturing.pipelines.market_intelligence_pipeline import (
            run_market_intelligence_pipeline,
        )
        # Step 1: 材料相場をチェック
        market_result = await run_market_intelligence_pipeline(
            company_id=company_id,
            input_data={
                "collection_type": "on_demand",
                "trigger": "margin_decline",
                "affected_processes": [a["process_type"] for a in margin_alerts],
            },
        )

        # Step 2: 材料相場が原因であれば値上げ提案を生成
        if market_result and market_result.material_impact:
            from workers.bpo.manufacturing.pipelines.pl_intelligence_pipeline import (
                generate_price_revision_proposal,
            )
            await generate_price_revision_proposal(
                company_id=company_id,
                margin_alerts=margin_alerts,
                material_impact=market_result.material_impact,
            )
            logger.info(f"chain: margin_decline_chain 完了 company_id={company_id}")

    except Exception as e:
        logger.error(f"chain: _run_margin_decline_chain 失敗（非致命的）: {e}")
